#include <stdio.h>

struct Book {
    char id[20];
    char title[100];
    char author[100];
    char status[20];
};

void saveBooks(struct Book books[], int totalBooks) {
    FILE *fp = fopen("books.txt", "w");

    if (fp == NULL) {
        printf("Error opening file!\n");
        return;
    }

    for (int i = 0; i < totalBooks; i++) {
        fprintf(fp, "%s,%s,%s,%s\n",
                books[i].id,
                books[i].title,
                books[i].author,
                books[i].status);
    }

    fclose(fp);
}

void loadBooks(struct Book books[], int *totalBooks) {
    FILE *fp = fopen("books.txt", "r");

    if (fp == NULL) {
        printf("No existing file found.\n");
        return;
    }

    while (fscanf(fp, "%[^,],%[^,],%[^,],%s\n",
                  books[*totalBooks].id,
                  books[*totalBooks].title,
                  books[*totalBooks].author,
                  books[*totalBooks].status) != EOF) {
        (*totalBooks)++;
    }

    fclose(fp);
}
